[docs]    def draw_test_vo_in(vo, target, query_id=False):
        if not target in ["KEGG", "WP", "Reactome"]:
            target = "KEGG"
        if target == "KEGG":
            res = PublicDatabase.search_kegg("b cell")
            path = res[1].load(query_entity_data=False)
            ids = [x.id for x in path.genes]
        elif target == "WP":
            res = PublicDatabase.search_wp("jak")
            path = res[1].load()
            ids = [x.props.get("GraphId") for x in path.flatten() if x._class == "DataNode"]
        else:
            # proxy = dict(https='socks5://127.0.0.1:1080', http='socks5://127.0.0.1:1080')
            res = PublicDatabase.search_reactome("t cell")  #, proxies=proxy)
            res[1].retrieve()
            # res[1].data = res[1].data.encode("utf-8")
            path = res[1].load(2)
            ids = ids = [x.id for x in path.nodes if x.type == "macromolecule" or x.type == "macromolecule multimer"]
        if query_id:
            return ids, path
        else:
            path.integrate(ids, [vo for _ in range(len(ids))])
            return path.draw()

[docs]    def test_connection(target="KEGG"):
        ids, path = MappingAPITest.draw_test_vo_in(None, target, True)
        vos = []
        for x in ids:
            vos.append(VisualizationOption(mouse_over=[
                Connection([Edge(x, width=5, line_style="dashed",
                                 line_color="red",
                                 target_style=ValueChanged({
                                            NP.BG_COLOR: rgb(255, 0, 0),
                                            NP.OPACITY: 1,
                                            NP.SCALE: 1,
                                            NP.COLOR: rgb(255, 255, 255)
                                        })
                                 ) for x in random.sample(ids, 5)])
            ]))
        path.integrate(ids, vos)
        return path.draw()
